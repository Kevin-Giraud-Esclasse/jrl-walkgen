SET (UNITTESTING_CXXFLAGS "-I${${PROJECT_NAME}_SOURCE_DIR}/include ${${PROJECT_NAME}_CXXFLAGS} ")

IF(UNIX)
  SET(UNITTESTING_LDFLAGS "${LIBDIR_KW}${${PROJECT_NAME}_BINARY_DIR}/src ${LIBDIR_RP}${${PROJECT_NAME}_BINARY_DIR}/src ${LIBINCLUSION_KW}${PROJECT_NAME} ${${PROJECT_NAME}_src_LDFLAGS}")
ENDIF(UNIX)

IF(WIN32)
  SET(UNITTESTING_LDFLAGS "${LIBDIR_KW}${${PROJECT_NAME}_BINARY_DIR}/src  ${LIBINCLUSION_KW}${PROJECT_NAME}-${PROJECT_VERSION}${LIBINCLUSION_EXT} ${${PROJECT_NAME}_src_LDFLAGS}")
ENDIF(WIN32)


MESSAGE(STATUS "UL: ${UNITTESTING_LDFLAGS}  ${UNITTESTING_CXXFLAGS}")

ADD_EXECUTABLE(	TestFootPrintPGInterface TestFootPrintPGInterface.cpp)
SET_TARGET_PROPERTIES(TestFootPrintPGInterface
	              PROPERTIES
	              COMPILE_FLAGS ${UNITTESTING_CXXFLAGS}
		      LINK_FLAGS ${UNITTESTING_LDFLAGS}
)
ADD_DEPENDENCIES (TestFootPrintPGInterface ${PROJECT_NAME})
